//---------Bloc 1 : En tête déclarative----------------------

#include <stdio.h> 
#include <stdlib.h>
#include <string.h>

	//Déclaration

const int LED = 10;     // la broche 10 devient LED
const int bouton = 4;   // la broche 4 devient Bouton

int B=5;

//--------Bloc 2 : Configuration initiale : Fonction Setup---------

	//Initialisation des broches, communications, variables,vitesse de communication...

void setup() {

  Serial.begin(9600);    	// Vitesse de communiction entre Arduino et le PC; 9600 bauds (bits/s)
  pinMode(LED, OUTPUT);  	// Configure la broche digital 10 en sortie
  pinMode(bouton, INPUT);  	// Configure la broche 4 en entrée 

}
//--------- Bloc 3 : Instructions exécutées en boucle : Fonction Loop()----------------
	// Fonction principale

void loop() {

  if (digitalRead(bouton) == 1 )  // Si bouton poussoir appuyé...
  
	//...on fait clignoter la LED  
  {     
  digitalWrite(LED, HIGH);
  delay(1000);
  digitalWrite(LED, LOW);
  delay(1000);
  } 
	// Sinon on éteint la LED  
 
  else    
  { 
  digitalWrite(LED, LOW);   // éteint la LED 
  }

}
